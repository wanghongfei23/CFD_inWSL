338real_t exact_initial_solution(real_t x, real_t /*t*/)
339{
340 // Example: exact solution for a periodic sine function
341 // u(x, t) = sin(2 * pi * (x - ct))

348

350 real_t x_c = 0.5Q;
351 return expq(-300.0Q * (x - x_c) * (x - x_c));

369 // real_t x_c = 0.5Q;

436void compute_rhs(GhostArray& rhs, GhostArray& u, GhostArray& flux, real_t dx)
437{
438 const real_t c = 1.0Q; // 波速
439 if (is_periodic)
440 u.fill_periodic();
441
442 // 1. 一阶插值计算界面通量 (i+1/2位置)
443// #pragma omp parallel for
444 for (int i : flux.full_range()) {
445 // 提取左值和右值
446 // real_t u_left = u[i - 1];
447 // real_t u_right = u[i];
448 std::array<real_t, 5> q_left = { u[i - 3], u[i - 2], u[i - 1], u[i], u[i + 1] };
449 std::array<real_t, 5> q_right = { u[i + 2], u[i + 1], u[i], u[i - 1], u[i - 2] };
450
451 //
452 // 使用高阶插值得到左值和右值
453
454 // 【王鸿飞】插值格式更换处
455 // real_t u_left = whf_TcnsN_A(q_left);
456 // real_t u_right = whf_TcnsN_A(q_right);
457 // real_t u_left = whf_TcnsN_myASF203_NoS(q_left);
458 // real_t u_right = whf_TcnsN_myASF203_NoS(q_right);
459 real_t u_left = whf_TcnsN_myASF213_NoS(q_left);
460 real_t u_right = whf_TcnsN_myASF213_NoS(q_right);
461 // real_t u_left = Linear_5th(q_left);
462 // real_t u_right = Linear_5th(q_right);
463
464 // 一阶插值:取左右平均值
465 // if (c > 0) {
466 // // c > 0, 使用前向差分
467 // flux[i] = c * u_left;
468 // } else {
469 // // c <= 0, 使用后向差分
470 // flux[i] = c * u_right;
471 // }
472
473 real_t F_left = c * u_left;
474 real_t F_right = c * u_right;
475 flux[i] = (F_left + F_right) / 2.0Q - (c / 2.0Q) * (u_right - u_left);
476 }
477 if (is_periodic)
478 flux.fill_periodic();
479 std::array<real_t, 3> w = { 3.0Q / 2.0Q, -3.0Q / 10.0Q, 1.0Q / 30.0Q };
480// std::array<real_t, 3> w = { 4.0Q / 3.0Q, -1.0Q / 6.0Q, 0.0Q };
481// std::array<real_t, 3> w = { 1.0Q, 0.0Q, 0.0Q };
482// 2. 二阶中心差分计算右端项
483// #pragma omp parallel for
484 for (int i : rhs.domain()) {
485 // 二阶差分: (flux[i+1] - flux[i]) / dx
486 // rhs[i] = -1.0Q / dx * (flux[i + 1] - flux[i]);
487 rhs[i] = -w[0] / dx * (flux[i + 1] - flux[i])
488 - w[1] / dx * (u[i + 1] - u[i - 1])
489 - w[2] / dx * (flux[i + 2] - flux[i - 1]);
490 }
491}

505void evolve(GhostArray& u, GhostArray& u_prev, GhostArray& flux, GhostArray& rhs,
506 real_t dt, real_t dx) // 添加dx参数
507{
508 // 阶段1
509 compute_rhs(rhs, u_prev, flux, dx); // 现在dx已定义
510// #pragma omp parallel for
511 for (int i : u.domain()) {
512 u[i] = u_prev[i] + dt * rhs[i];
513 }
514
515 // 阶段2
516 GhostArray temp(u);
517 compute_rhs(rhs, u, flux, dx); // 传递dx
518// #pragma omp parallel for
519 for (int i : u.domain()) {
520 temp[i] = 0.75Q * u_prev[i] + 0.25Q * (u[i] + dt * rhs[i]);
521 }
522
523 // 阶段3
524 compute_rhs(rhs, temp, flux, dx); // 传递dx
525// #pragma omp parallel for
526 for (int i : u.domain()) {
527 u[i] = (1.0Q / 3.0Q) * u_prev[i] + (2.0Q / 3.0Q) * (temp[i] + dt * rhs[i]);
528 }
529}

540inline real_t L1_norm_error(const GhostArray& u, const GhostArray& x, real_t t)
541{
542 real_t error = 0.0Q;
543 for (int i : u.domain()) {
544 real_t xi = x[i];
545 error += fabsq(u[i] - exact_solution(xi, t));
546 }
547 return error / u.get_domain_size();
548}

559inline real_t L2_norm_error(const GhostArray& u, const GhostArray& x, real_t t)
560{
561 real_t error = 0.0Q;
562 for (int i : u.domain()) {
563 real_t xi = x[i];
564 error += powq(u[i] - exact_solution(xi, t), 2);
565 }
566 return sqrtq(error / u.get_domain_size());
567}
