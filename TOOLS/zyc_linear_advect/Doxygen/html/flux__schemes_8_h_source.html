80 1.0/4.0 * powq(1.0 * q[0] - 4.0 * q[1] + 3.0 * q[2], 2);

82 1.0/4.0 * powq(1.0 * q[1] + 0.0 * q[2] - 1.0 * q[3], 2);
83 beta[2] = 1.0/1.0 * powq(1.0 * q[2] - 2.0 * q[3] + 1.0 * q[4], 2) +
84 1.0/4.0 * powq(3.0 * q[2] - 4.0 * q[3] + 1.0 * q[4], 2);
85 // 全局光滑因子τ
86 real_t tau = std::abs(beta[2] - beta[0]);
87
88 // 计算自适应阈值CT_A
89 real_t alpha1 = 10.0, alpha2 = 5.0; //自适应参数
90 real_t xi = 1e-3, C_r = 0.24; //小参数和切换参数
91 real_t epsilon_A = (0.9*C_r)/(1.0 - 0.9*C_r)*xi*xi; //自适应参数ε_A
92
93 // 计算差分值
94 std::array<real_t, 4> delta_q;
95 delta_q[0] = q[0] - q[1];
96 delta_q[1] = q[1] - q[2];
97 delta_q[2] = q[2] - q[3];
98 delta_q[3] = q[3] - q[4];
99
100 // 计算η值（用于检测解的局部特征）
101 std::array<real_t, 3> eta;
102 eta[0] = (std::abs(2.0 * delta_q[0] * delta_q[1]) + epsilon_A)
103 / (delta_q[0] * delta_q[0] + delta_q[1] * delta_q[1] + epsilon_A);
104 eta[1] = (std::abs(2.0 * delta_q[1] * delta_q[2]) + epsilon_A)
105 / (delta_q[1] * delta_q[1] + delta_q[2] * delta_q[2] + epsilon_A);
106 eta[2] = (std::abs(2.0 * delta_q[2] * delta_q[3]) + epsilon_A)
107 / (delta_q[2] * delta_q[2] + delta_q[3] * delta_q[3] + epsilon_A);
108
109 // 计算η_min值（最小的η值）
110 real_t eta_min = std::min({eta[0],eta[1],eta[2]});
111
112 // 计算自适应函数参数
113 real_t m = 1.0 - std::min(1.0Q, eta_min/C_r);
114 real_t g_m = (1 - m)*(1 - m)*(1 - m)*(1 - m)*(1 + 4*m);
115
116 // 计算beta_A参数
117 int beta_A = (int)floorq(alpha1 - alpha2*(1 - g_m));
118
119 // 自适应阈值参数CT_A
120 real_t CT_A;
121
122 // 求光滑度量gamma（用于构造截止函数）
123 const real_t C = 1.0;
124 const real_t eps = 1e-40;
125
126 std::array<real_t, 3> gamma;
127
128 gamma[0] = C + tau/(beta[0] + eps);
129 gamma[1] = C + tau/(beta[1] + eps);
130 gamma[2] = C + tau/(beta[2] + eps);
131
132 // 对gamma进行幂运算处理
133 gamma[0] = powq(gamma[0], 2);
134 gamma[1] = powq(gamma[1], 2);
135 gamma[2] = powq(gamma[2], 2);
136
137 gamma[0] = powq(gamma[0], 3);
138 gamma[1] = powq(gamma[1], 3);
139 gamma[2] = powq(gamma[2], 3);
140
141 // 计算gamma总和
142 real_t gamma_sum = gamma[0] + gamma[1] + gamma[2];
143
144 // 根据beta_A值确定CT_A
145 switch(beta_A) {
146 case 5: CT_A = 1e-5;
147 break;
148 case 6: CT_A = 1e-6;
149 break;
150 case 7: CT_A = 1e-7;
151 break;
152 case 8: CT_A = 1e-8;
153 break;
154 case 9: CT_A = 1e-9;
155 break;
156 case 10: CT_A = 1e-10;
157 break;
158 }
159
160 // 计算判定参数rr
161 real_t rr;
162 rr = CT_A*gamma_sum;
163 // 构造截止函数
164 unsigned short flag = 0;
165 if (gamma[0] < rr) flag += 1;
166 if (gamma[1] < rr) flag += 2;
167 if (gamma[2] < rr) flag += 4;

169 // 根据flag值进行插值
170 switch (flag) {
171 case 0: // 111
172 return 3.0/128.0 * q[0] - 5.0/32.0 * q[1] + 45.0/64.0 * q[2] + 15.0/32.0 * q[3] - 5.0/128.0 * q[4];
173 case 1: // 011
174 return -1.0/16.0 * q[1] + 9.0/16.0 * q[2] + 9.0/16.0 * q[3] - 1.0/16.0 * q[4];
175 case 2: // 101
176 case 3: // 001
177 return 3.0/8.0 * q[2] + 3.0/4.0 * q[3] - 1.0/8.0 * q[4];
178 case 4: // 110
179 return 1.0/16.0 * q[0] - 5.0/16.0 * q[1] + 15.0/16.0 * q[2] + 5.0/16.0 * q[3];

200 // 解包输入值
201 real_t q_im2 = q[0]; // q_{i-2}
202 real_t q_im1 = q[1]; // q_{i-1}

207 // 计算差分值
208 real_t dq_im3_2 = q_im1 - q_im2; // Δq_{i-3/2}
209 real_t dq_im1_2 = q_i - q_im1; // Δq_{i-1/2}
210 real_t dq_ip1_2 = q_ip1 - q_i; // Δq_{i+1/2}
211 real_t dq_ip3_2 = q_ip2 - q_ip1; // Δq_{i+3/2}
212
213 // 计算局部光滑因子
214 real_t beta0 = powq(q_im2 - 2*q_im1 + q_i, 2) +
215 0.25 * powq(q_im2 - 4*q_im1 + 3*q_i, 2);

221 0.25 * powq(3*q_i - 4*q_ip1 + q_ip2, 2);
222
223 // 计算全局光滑因子
224 real_t tau = std::abs(beta2 - beta0);
225
226 // 计算η值
227 real_t eta_im1 = (std::abs(2*dq_im1_2*dq_im3_2) + epsilon_A) /
228 (powq(dq_im1_2, 2) + powq(dq_im3_2, 2) + epsilon_A);
229
230 real_t eta_i = (std::abs(2*dq_ip1_2*dq_im1_2) + epsilon_A) /
231 (powq(dq_ip1_2, 2) + powq(dq_im1_2, 2) + epsilon_A);
232
233 real_t eta_ip1 = (std::abs(2*dq_ip3_2*dq_ip1_2) + epsilon_A) /
234 (powq(dq_ip3_2, 2) + powq(dq_ip1_2, 2) + epsilon_A);
235
236 real_t eta_min = std::min({eta_im1, eta_i, eta_ip1});
237
238 // 计算m参数
239 real_t min = std::min(1.0Q, eta_min / Cr);
240 real_t m = 1.0Q - min;
241
242 // 计算C_T_prime（自适应阈值参数）
243 real_t C_T_prime = -0.452662*m*m*m + 0.590811*m*m - 0.012189*m + 0.022010;
244
245 // 确定β_j
246 real_t beta_j = std::min({beta0, beta1, beta2});
247
248 // 计算δ_k
249 int delta0 = (beta_j * tau < (C_T_prime * tau - C * (1 - C_T_prime) * beta_j) * beta0) ? 0 : 1;
250 int delta1 = (beta_j * tau < (C_T_prime * tau - C * (1 - C_T_prime) * beta_j) * beta1) ? 0 : 1;
251 int delta2 = (beta_j * tau < (C_T_prime * tau - C * (1 - C_T_prime) * beta_j) * beta2) ? 0 : 1;
252
253 // 根据δ值选择插值系数
254 real_t a0, a1, a2, a3, a4;
255
256 if (delta0 == 1 && delta1 == 1 && delta2 == 1) {
257 a0 = 3.0/128; a1 = -5.0/32; a2 = 45.0/64; a3 = 15.0/32; a4 = -5.0/128;
258 }
259 else if (delta0 == 0 && delta1 == 1 && delta2 == 1) {
260 a0 = 0; a1 = -1.0/16; a2 = 9.0/16; a3 = 9.0/16; a4 = -1.0/16;
261 }
262 else if ((delta0 == 1 && delta1 == 0 && delta2 == 1) ||
263 (delta0 == 0 && delta1 == 0 && delta2 == 1)) {
264 a0 = 0; a1 = 0; a2 = 3.0/8; a3 = 3.0/4; a4 = -1.0/8;
265 }
266 else if (delta0 == 1 && delta1 == 1 && delta2 == 0) {
267 a0 = 1.0/16; a1 = -5.0/16; a2 = 15.0/16; a3 = 5.0/16; a4 = 0;
268 }
269 else if (delta0 == 0 && delta1 == 1 && delta2 == 0) {
270 a0 = 0; a1 = -1.0/8; a2 = 3.0/4; a3 = 3.0/8; a4 = 0;

273 a0 = 3.0/8; a1 = -5.0/4; a2 = 15.0/8; a3 = 0; a4 = 0;

275 else { // (0,0,0)
276 a0 = 0; a1 = 0; a2 = 1; a3 = 0; a4 = 0;
277 }
278
279 // 计算最终插值结果
280 return a0*q_im2 + a1*q_im1 + a2*q_i + a3*q_ip1 + a4*q_ip2;
