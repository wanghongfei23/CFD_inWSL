542void SpaceDis::calFlux()
543{
544 // 获取左边界和右边界ghost cell的数量，用于确定计算范围
545 int fGhostL = fBndL->getN(), fGhostR = fBndR->getN();
546 // 将通量数组初始化为零
547 flux_d->setZeros();
548 // 在包括ghost cells在内的所有单元格面上计算通量
549 // 计算范围从最左边的ghost cell面(索引为0-fGhostL)到最右边的ghost cell面(索引为nHalf+fGhostR-1)
550 for (int i = 0 - fGhostL; i < nHalf + fGhostR; i++) {
551 // 使用函数指针调用相应方程类型的通量计算方法
552 // 根据问题类型(calTypeFlux)调用对应的通量计算函数
553 (this->*calTypeFlux)(i);
554 }
