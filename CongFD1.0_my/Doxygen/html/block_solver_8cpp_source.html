78void BlockSolver::RK3_SSP(real dt)
79{
80
81 Data tempdata(*cons);
82 int n=cons->size();
83
84
85 //third order RK
86 //stage 1
87 rhs->setZeros();
88 eqn->consToPrim();
89 bnds->update();
90 spDis->rhsSolve();
91 sourceTerm->calSource();
92 // cgnsIO.BlockCgnsOutput(block,info);
93 // cgnsIO.solCgnsOutput(rhs,info);
94 #pragma omp parallel for
95 for(int i=0;i<n;i++)
96 {
97 (*cons)[i]=tempdata[i]-dt*(*rhs)[i];
98 }
99 info->t+=dt;
100
101 //stage 2
102 rhs->setZeros();
103 eqn->consToPrim();
104 bnds->update();
105 spDis->rhsSolve();
106 sourceTerm->calSource();
107 #pragma omp parallel for
108 for(int i=0;i<n;i++)
109 {
110 (*cons)[i]=0.75*tempdata[i]-0.25*dt*(*rhs)[i]+0.25*(*cons)[i];
111 }
112 info->t-=dt/2;
113
114 //stage 3
115 rhs->setZeros();
116 eqn->consToPrim();
117 bnds->update();
118 spDis->rhsSolve();
119 sourceTerm->calSource();
120 #pragma omp parallel for
121 for(int i=0;i<n;i++)
122 {
123 (*cons)[i]=1.0/3.0*tempdata[i]-2.0/3.0*dt*(*rhs)[i]+2.0/3.0*(*cons)[i];
124 }
125 info->t-=dt/2;
126}

137void BlockSolver::DTS_Euler(real dt)
138{
139
140
141 int n=cons->size();
142 Data tempdata(*cons);
143 Data tempRhs(*cons);
144 real maxres=0;
145
146 int imStep=0;
147 do
148 {
149
150 auto dtau=calLocalCFL();
151 rhs->setZeros();
152 eqn->consToPrim();
153 bnds->update();
154 spDis->rhsSolve();
155 sourceTerm->calSource();
156 // cgnsIO.BlockCgnsOutput(block,info);
157 // cgnsIO.solCgnsOutput(rhs,info);
158 tempRhs.setZeros();
159 #pragma omp parallel for
160 for(int i=0;i<n;i++)
161 {
162 real coef=dtau[i]/(dt+dtau[i]);
163 tempRhs[i]=(tempdata[i]-(*cons)[i]-dt*(*rhs)[i])*coef;
164 (*cons)[i]=(*cons)[i]+tempRhs[i];
165 }
166 maxres=tempRhs.getLinf(0);
167 std::cout<<std::format("time = {:.4f} dt={:.10f} imStep={} rhoL2 = {} \n"
168 ,info->t,dt,imStep,maxres);
169 }while(imStep++<info->maxImplicitStep && maxres>1e-7);

272void BlockSolver::stepsLoopCFL()
273{
274 const real eps=1e-15; // 定义一个极小值，用于浮点数比较
275 while(info->step<info->endStep) // 当前步数小于结束步数时继续循环
276 {
277 // 当输出时间接近0时，输出网格和流场数据
278 if(info->outputT<eps) // 判断是否需要输出结果（输出时间接近0）
279 {
280 outputGrid(); // 输出网格信息到文件
281 outputPrim(); // 输出原始变量到文件
282 }
283
284 auto dt=getTimeIntervalExplicit(); // 根据CFL条件自动计算时间步长
285 // dt=info->dt;
286
287 // 如果下一个时间步会超过输出间隔，则调整时间步长以精确匹配输出时刻
288 if(info->outputT+dt>info->outputDt) // 判断加上当前步长是否会超过输出时间间隔
289 {
290 dt=info->outputDt-info->outputT; // 调整时间步长，使其恰好到达输出时刻
291 info->outputT=0.0; // 重置输出时间计数器
292 info->step++; // 步数增加1
293 }
294 else info->outputT+=dt; // 否则更新输出时间计数器
295
296 // 执行求解并计时
297 auto start = std::chrono::high_resolution_clock::now(); // 记录求解开始时间
298 solve(dt); // 执行一次时间步长为dt的求解
299 auto stop = std::chrono::high_resolution_clock::now(); // 记录求解结束时间
300 auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(stop - start).count(); // 计算求解耗时（毫秒）
301 timesss+=duration; // 累加总求解时间
302 timesteps++; // 时间步数计数器增加
303
304 // 输出当前时间、时间步长和密度Linf范数
305 std::cout<<std::format("time = {:.4f} dt={:.10f} rhoLinf = {:.4f} \n",info->t,dt,rhs->getLinf(0)); // 输出当前时间、时间步长和守恒变量第0个分量的Linf范数
306 }
307 // 求解完成后输出最终结果
308 outputGrid(); // 求解完成后输出最终网格信息
309 outputPrim(); // 求解完成后输出最终原始变量
310}

335std::vector<real> BlockSolver::calLocalCFL()
336{
337
338 real dt,lambda=0;
339
340 Data* prim=eqn->getPrim();
341 int n=prim->getN();
342 std::vector<real> res(n);
343 eqn->consToPrim();
344 #pragma omp parallel for reduction(max:lambda)
345 for(int i=0;i<n;i++)
346 {
347 real iLambda;
348 real dh=block->getMinDh(i);
349 real RT=std::sqrt(GAMMA*(*prim)(i,2)/(*prim)(i,0))+std::abs((*prim)(i,1));
350 if(info->dim==2) iLambda=(std::sqrt(GAMMA*(*prim)(i,3)/(*prim)(i,0))+std::abs((*prim)(i,1))+std::abs((*prim)(i,2)))/dh;
351 else if(info->dim==1) iLambda=(std::sqrt(GAMMA*(*prim)(i,2)/(*prim)(i,0))+std::abs((*prim)(i,1)))/dh;
352 res[i]=info->implicitCFL/iLambda;
353 }
354 if(info->eqType!=EULER)
355 { std::cout<<"Block Solver error: CFL Loop with not EULER solver\n";}
356 return res;

418real BlockSolver::getTimeIntervalExplicit()
419{
420 real dt,lambda=0;
421 int dim=info->dim;
422 if(info->eqType==EULER)
423 {
424 Data* prim=eqn->getPrim();
425 int n=prim->getN();
426 eqn->consToPrim();
427 std::vector<real> lambdas(dim);
428 for(int idim=0;idim<dim;idim++)
429 {
430 real maxLambda=0;
431 #pragma omp parallel for reduction(max:maxLambda)
432 for(int i=0;i<n;i++)
433 {
434 real iLambda;
435 auto dhs=block->getCellInterval(i);
436 real dh=info->constH?info->geth(idim):dhs.at(idim);
437 iLambda=(std::sqrt(GAMMA*(*prim)(i,dim+1)/(*prim)(i,0))+std::abs((*prim)(i,idim+1)))/dh;
438 maxLambda=std::max(maxLambda,iLambda);
439 }
440 lambdas[idim]=maxLambda;
441 }
442 lambda=0;
443 for(auto iLambda:lambdas) lambda+=iLambda;
444
445 }
446 else
447 { std::cout<<"Block Solver error: CFL Loop with not EULER solver\n"; lambda=1;}
448 dt=info->CFL/lambda;
449 return dt;
