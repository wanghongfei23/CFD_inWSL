79void BlockSolver::RK3_SSP(real dt)
80{
81
82 Data tempdata(*cons);
83 int n=cons->size();
84
85
86 //third order RK
87 //stage 1
88 rhs->setZeros();
89 eqn->consToPrim();
90 bnds->update();
91 spDis->rhsSolve();
92 sourceTerm->calSource();
93 // cgnsIO.BlockCgnsOutput(block,info);
94 // cgnsIO.solCgnsOutput(rhs,info);
95 #pragma omp parallel for
96 for(int i=0;i<n;i++)
97 {
98 (*cons)[i]=tempdata[i]-dt*(*rhs)[i];
99 }
100 info->t+=dt;
101
102 //stage 2
103 rhs->setZeros();
104 eqn->consToPrim();
105 bnds->update();
106 spDis->rhsSolve();
107 sourceTerm->calSource();
108 #pragma omp parallel for
109 for(int i=0;i<n;i++)
110 {
111 (*cons)[i]=0.75*tempdata[i]-0.25*dt*(*rhs)[i]+0.25*(*cons)[i];
112 }
113 info->t-=dt/2;
114
115 //stage 3
116 rhs->setZeros();
117 eqn->consToPrim();
118 bnds->update();
119 spDis->rhsSolve();
120 sourceTerm->calSource();
121 #pragma omp parallel for
122 for(int i=0;i<n;i++)
123 {
124 (*cons)[i]=1.0/3.0*tempdata[i]-2.0/3.0*dt*(*rhs)[i]+2.0/3.0*(*cons)[i];
125 }
126 info->t-=dt/2;
127}

138void BlockSolver::DTS_Euler(real dt)
139{
140
141
142 int n=cons->size();
143 Data tempdata(*cons);
144 Data tempRhs(*cons);
145 real maxres=0;
146
147 int imStep=0;
148 do
149 {
150
151 auto dtau=calLocalCFL();
152 rhs->setZeros();
153 eqn->consToPrim();
154 bnds->update();
155 spDis->rhsSolve();
156 sourceTerm->calSource();
157 // cgnsIO.BlockCgnsOutput(block,info);
158 // cgnsIO.solCgnsOutput(rhs,info);
159 tempRhs.setZeros();
160 #pragma omp parallel for
161 for(int i=0;i<n;i++)
162 {
163 real coef=dtau[i]/(dt+dtau[i]);
164 tempRhs[i]=(tempdata[i]-(*cons)[i]-dt*(*rhs)[i])*coef;
165 (*cons)[i]=(*cons)[i]+tempRhs[i];
166 }
167 maxres=tempRhs.getLinf(0);
168 // 修复：使用传统的iostream方式替代std::format以避免编译错误
169 std::cout << "time = " << std::fixed << std::setprecision(4) << info->t
170 << " dt=" << std::setprecision(10) << dt
171 << " imStep=" << imStep
172 << " rhoL2 = " << std::setprecision(4) << maxres << " \n";
173 }while(imStep++<info->maxImplicitStep && maxres>1e-7);
174

176}

374std::vector<real> BlockSolver::calLocalCFL()
375{
376
377 real dt,lambda=0;
378
379 Data* prim=eqn->getPrim();
380 int n=prim->getN();
381 std::vector<real> res(n);
382 eqn->consToPrim();
383 #pragma omp parallel for reduction(max:lambda)
384 for(int i=0;i<n;i++)
385 {
386 real iLambda;
387 real dh=block->getMinDh(i);
388 real RT=std::sqrt(GAMMA*(*prim)(i,2)/(*prim)(i,0))+std::abs((*prim)(i,1));
389 if(info->dim==2) iLambda=(std::sqrt(GAMMA*(*prim)(i,3)/(*prim)(i,0))+std::abs((*prim)(i,1))+std::abs((*prim)(i,2)))/dh;
390 else if(info->dim==1) iLambda=(std::sqrt(GAMMA*(*prim)(i,2)/(*prim)(i,0))+std::abs((*prim)(i,1)))/dh;
391 res[i]=info->implicitCFL/iLambda;
392 }
393 if(info->eqType!=EULER)
394 { std::cout<<"Block Solver error: CFL Loop with not EULER solver\n";}

457real BlockSolver::getTimeIntervalExplicit()
458{
459 real dt,lambda=0;
460 int dim=info->dim;
461 if(info->eqType==EULER)
462 {
463 Data* prim=eqn->getPrim();
464 int n=prim->getN();
465 eqn->consToPrim();
466 std::vector<real> lambdas(dim);
467 for(int idim=0;idim<dim;idim++)
468 {
469 real maxLambda=0;
470 #pragma omp parallel for reduction(max:maxLambda)
471 for(int i=0;i<n;i++)
472 {
473 real iLambda;
474 auto dhs=block->getCellInterval(i);
475 real dh=info->constH?info->geth(idim):dhs.at(idim);
476 iLambda=(std::sqrt(GAMMA*(*prim)(i,dim+1)/(*prim)(i,0))+std::abs((*prim)(i,idim+1)))/dh;
477 maxLambda=std::max(maxLambda,iLambda);
478 }
479 lambdas[idim]=maxLambda;
480 }
481 lambda=0;
482 for(auto iLambda:lambdas) lambda+=iLambda;
483
484 }
485 else
486 { std::cout<<"Block Solver error: CFL Loop with not EULER solver\n"; lambda=1;}
487 dt=info->CFL/lambda;
488 return dt;
