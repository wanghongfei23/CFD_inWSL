79void BlockSolver::RK3_SSP(real dt)
80{
81
82 Data tempdata(*cons);
83 int n=cons->size();
84
85
86 // third order RK
87 // stage 1
88 // 将右端项(rhs)重置为零，为新一轮计算做准备
89 rhs->setZeros();
90 // 将守恒变量转换为原始变量（密度、速度、压力等），用于后续的通量计算
91 eqn->consToPrim();
92 // 更新所有边界条件，确保边界上的值反映最新的内部流场状态
93 bnds->update();
94 // 计算空间离散项，即对流项的数值导数，贡献到右端项中
95 spDis->rhsSolve();
96 // 计算物理源项（如体积力、化学反应等）并将其贡献添加到右端项中
97 sourceTerm->calSource();
98 // cgnsIO.BlockCgnsOutput(block,info);
99 // cgnsIO.solCgnsOutput(rhs,info);
100 #pragma omp parallel for
101 for(int i=0;i<n;i++)
102 {
103 (*cons)[i]=tempdata[i]-dt*(*rhs)[i];
104 }
105 info->t+=dt;
106
107 //stage 2
108 // 将右端项重置为零，开始新的时间步计算
109 rhs->setZeros();
110 // 将守恒变量转换为原始变量，用于计算通量
111 eqn->consToPrim();
112 // 更新边界条件，确保边界值是最新的
113 bnds->update();
114 // 计算空间离散项，得到对流项的数值导数贡献
115 spDis->rhsSolve();
116 // 计算源项（如体积力、能量源等）并添加到右端项中
117 sourceTerm->calSource();
118 #pragma omp parallel for
119 for(int i=0;i<n;i++)
120 {
121 (*cons)[i]=0.75*tempdata[i]-0.25*dt*(*rhs)[i]+0.25*(*cons)[i];
122 }
123 info->t-=dt/2;
124
125 //stage 3
126 rhs->setZeros();
127 eqn->consToPrim();
128 bnds->update();
129 spDis->rhsSolve();
130 sourceTerm->calSource();
131 #pragma omp parallel for
132 for(int i=0;i<n;i++)
133 {
134 (*cons)[i]=1.0/3.0*tempdata[i]-2.0/3.0*dt*(*rhs)[i]+2.0/3.0*(*cons)[i];
135 }
136 info->t-=dt/2;
137}

148void BlockSolver::DTS_Euler(real dt)
149{
150
151
152 int n=cons->size();
153 Data tempdata(*cons);
154 Data tempRhs(*cons);
155 real maxres=0;
156
157 int imStep=0;
158 do
159 {
160
161 auto dtau=calLocalCFL();
162 // 将右端项(rhs)重置为零，开始新的隐式迭代计算
163 rhs->setZeros();
164 // 将当前守恒变量转换为原始变量，用于计算通量和雅可比矩阵
165 eqn->consToPrim();
166 // 更新边界条件，确保在迭代过程中边界值保持最新
167 bnds->update();
168 // 计算空间离散项（对流项的数值导数），贡献到右端项中
169 spDis->rhsSolve();
170 // 计算源项（如粘性、热传导等）并添加到右端项中
171 sourceTerm->calSource();
172 // cgnsIO.BlockCgnsOutput(block,info);
173 // cgnsIO.solCgnsOutput(rhs,info);
174 tempRhs.setZeros();
175 #pragma omp parallel for
176 for(int i=0;i<n;i++)
177 {
178 real coef=dtau[i]/(dt+dtau[i]);
179 tempRhs[i]=(tempdata[i]-(*cons)[i]-dt*(*rhs)[i])*coef;
180 (*cons)[i]=(*cons)[i]+tempRhs[i];
181 }
182 maxres=tempRhs.getLinf(0);
183 // 修复：使用传统的iostream方式替代std::format以避免编译错误
184 std::cout << "time = " << std::fixed << std::setprecision(4) << info->t
185 << " dt=" << std::setprecision(10) << dt
186 << " imStep=" << imStep
187 << " rhoL2 = " << std::setprecision(4) << maxres << " \n";
188 }while(imStep++<info->maxImplicitStep && maxres>1e-7);
189
190
191}

389std::vector<real> BlockSolver::calLocalCFL()
390{
391
392 real dt,lambda=0;
393
394 Data* prim = eqn->getPrim();
395 int n = prim->getN();
396 std::vector<real> res(n);
397 // 将守恒变量转换为原始变量，以便计算局部CFL数
398 eqn->consToPrim();
399 // 并行计算每个网格点的局部CFL时间步长
400 #pragma omp parallel for reduction(max:lambda)
401 for(int i=0;i<n;i++)
402 {
403 real iLambda;
404 real dh=block->getMinDh(i);
405 real RT=std::sqrt(GAMMA*(*prim)(i,2)/(*prim)(i,0))+std::abs((*prim)(i,1));
406 if(info->dim==2) iLambda=(std::sqrt(GAMMA*(*prim)(i,3)/(*prim)(i,0))+std::abs((*prim)(i,1))+std::abs((*prim)(i,2)))/dh;
407 else if(info->dim==1) iLambda=(std::sqrt(GAMMA*(*prim)(i,2)/(*prim)(i,0))+std::abs((*prim)(i,1)))/dh;
408 res[i]=info->implicitCFL/iLambda;
409 }
410 if(info->eqType!=EULER)
411 { std::cout<<"Block Solver error: CFL Loop with not EULER solver\n";}
412 return res;

422void BlockSolver::stepsLoopDTS()
423{
424 const real eps=1e-15;
425 while(info->step<info->endStep)
426 {
427 if(info->outputT<eps)
428 {
429 outputGrid();
430 outputPrim();
431 }
432
433 real dt, lambda = 0;
434 Data* prim = eqn->getPrim();
435 int n = prim->getN();
436 // 将守恒变量转换为原始变量，以便计算局部波速
437 eqn->consToPrim();
438 // 并行计算全局最大特征值（lambda），用于确定稳定的时间步长
439 #pragma omp parallel for reduction(max:lambda)
440 for(int i=0;i<n;i++)
441 {
442 real iLambda;
443 real dh=block->getMinDh(i);
444 real RT=std::sqrt(GAMMA*(*prim)(i,2)/(*prim)(i,0))+std::abs((*prim)(i,1));
445 if(info->dim==2) iLambda=(std::sqrt(GAMMA*(*prim)(i,3)/(*prim)(i,0))+std::abs((*prim)(i,1))+std::abs((*prim)(i,2)))/dh;
446 else if(info->dim==1) iLambda=(std::sqrt(GAMMA*(*prim)(i,2)/(*prim)(i,0))+std::abs((*prim)(i,1)))/dh;
447 lambda=std::max(lambda,iLambda);
448 }
449 if(info->eqType!=EULER)
450 { std::cout<<"Block Solver error: CFL Loop with not EULER solver\n"; lambda=1;}
451 dt=info->CFL/lambda;
452 // dt=info->dt;
453 if(info->outputT+dt>info->outputDt)
454 {
455 dt=info->outputDt-info->outputT;
456 info->outputT=0.0;
457 info->step++;
458 }
459 else info->outputT+=dt;
460
461 DTS_Euler(dt);
462 info->t+=dt;
463 //std::cout<<std::format("time = {:.4f} dt={:.10f} rhoL2 = {:.4f} \n",info->t,dt,rhs->getL2(0));
464 }
465 outputGrid();
466 outputPrim();
467}

476real BlockSolver::getTimeIntervalExplicit()
477{
478 real dt,lambda=0;
479 int dim=info->dim;
480 if(info->eqType == EULER)
481 {
482 Data* prim = eqn->getPrim();
483 int n = prim->getN();
484 // 将守恒变量转换为原始变量，用于计算声速和流动速度
485 eqn->consToPrim();
486 std::vector<real> lambdas(dim);
487 for(int idim=0;idim<dim;idim++)
488 {
489 real maxLambda=0;
490 #pragma omp parallel for reduction(max:maxLambda)
491 for(int i=0;i<n;i++)
492 {
493 real iLambda;
494 auto dhs=block->getCellInterval(i);
495 real dh=info->constH?info->geth(idim):dhs.at(idim);
496 iLambda=(std::sqrt(GAMMA*(*prim)(i,dim+1)/(*prim)(i,0))+std::abs((*prim)(i,idim+1)))/dh;
497 maxLambda=std::max(maxLambda,iLambda);
498 }
499 lambdas[idim]=maxLambda;
500 }
501 lambda=0;
502 for(auto iLambda:lambdas) lambda+=iLambda;
503
504 }
505 else
506 { std::cout<<"Block Solver error: CFL Loop with not EULER solver\n"; lambda=1;}
507 dt=info->CFL/lambda;
508 return dt;
509}
