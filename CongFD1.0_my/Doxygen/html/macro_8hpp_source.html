180constexpr std::array<int, 2> calOffset(int idim, int i, int j, std::array<int, 3> iMax)
181{
182 std::array<int, 3> offsets { 1, iMax[0], iMax[0] * iMax[1] };
183 std::array<int, 2> res;
184 if (idim == 1) {
185 res[0] = i * offsets[1] + j * offsets[2]; // i0
186 res[1] = offsets[0]; // offset
187 } else if (idim == 2) {
188 res[0] = i * offsets[0] + j * offsets[2]; // i0
189 res[1] = offsets[1]; // offset
190 } else if (idim == 3) {
191 res[0] = i * offsets[0] + j * offsets[1]; // i0
192 res[1] = offsets[2]; // offset
193 }
194 return res;
195}

205constexpr std::array<int, 2> calOffsetInverse(int idim, int i, int j, std::array<int, 3> iMax)
206{
207 std::array<int, 3> offsets { 1, iMax[0], iMax[0] * iMax[1] };
208 std::array<int, 2> res;
209 if (idim == 1) {
210 res[0] = i * offsets[1] + j * offsets[2] + (iMax[0] - 1) * offsets[0]; // i0
211 res[1] = -offsets[0]; // offset
212 } else if (idim == 2) {
213 res[0] = i * offsets[0] + j * offsets[2] + (iMax[1] - 1) * offsets[1]; // i0
214 res[1] = -offsets[1]; // offset
215 } else if (idim == 3) {
216 res[0] = i * offsets[0] + j * offsets[1] + (iMax[2] - 1) * offsets[2]; // i0
217 res[1] = -offsets[2]; // offset
218 }
