228{
229 // 字符串拼接，包含空间离散方法、插值方法、算例信息和当前时间
230 std::string caseName = "unknown";
231
232 if (dim == 1) {
233 auto it = exampleStr1D.find(nCase);
234 if (it != exampleStr1D.end()) {
235 caseName = it->second;
236 }
237 } else if (dim == 2) {
238 auto it = exampleStr2D.find(nCase);
239 if (it != exampleStr2D.end()) {
240 caseName = it->second;
241 }
242 }
243
244 // 添加网格信息到文件名
245 std::string gridInfo = "";
246 if (dim == 1) {
247 gridInfo = std::to_string(iMax[0]);

250 } else if (dim == 3) {
251 gridInfo = std::to_string(iMax[0]) + "x" + std::to_string(iMax[1]) + "x" + std::to_string(iMax[2]);
252 }
253
254 return caseName + " - " + disStr[interMethod] + " - " + gridInfo + " - " + std::format("t={:.4f}.cgns",t);
