15void Block::outputCgns()
16{
17 cgsize_t isize[3][dim];
18 int ni,nj,nk,i,j,k;
19 int index_file,icelldim,iphysdim,index_base;
20 int index_zone,index_coord;
21 char basename[33],zonename[33];
22
23/* create gridpoints for simple example: */
24 std::vector<real> x;
25 x.reserve(nVer);
26
27/* WRITE X, Y, Z GRID POINTS TO CGNS FILE */
28/* open CGNS file for write */
29 if (cg_open("grid_c.cgns",CG_MODE_WRITE,&index_file)) cg_error_exit();
30/* create base (user can give any name) */
31 strcpy(basename,"Base");
32 icelldim=dim;
33 iphysdim=dim;
34 cg_base_write(index_file,basename,icelldim,iphysdim,&index_base);
35/* define zone name (user can give any name) */
36 strcpy(zonename,"Zone 1");
37 for (int idim = 0; idim < dim; idim++)
38 {
39 /* vertex size */
40 isize[0][idim]=iMax[idim];
41 /* cell size */
42 isize[1][idim]=isize[0][idim]-1;
43 /* boundary vertex size (always zero for structured grids) */
44 isize[2][idim]=0;
45 }
46
47
48/* create zone */
49 cg_zone_write(index_file,index_base,zonename,*isize,CGNS_ENUMV(Structured),&index_zone);
50/* write grid coordinates (user must use SIDS-standard names here) */
51 for (int idim = 0; idim < dim; idim++)
52 {
53 std::string name=(idim==0)? "CoordinateX":(idim==1)? "CoordinateY":"CoordinateZ";
54 for (int i=0 ; i<nVer ; i++ ) x[i]=coorVer(i,idim);
55 cg_coord_write(index_file,index_base,index_zone,CGNS_ENUMV(RealDouble),
56 name.c_str(),
57 x.data(),&index_coord);
58 }
59
60
61/* close CGNS file */
62 cg_close(index_file);
63}
