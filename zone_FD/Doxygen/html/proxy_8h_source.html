107template <class T>
108consteval bool has_copyability(constraint_level level) {
109 switch (level) {
110 case constraint_level::none: return true;
111 case constraint_level::nontrivial: return std::is_copy_constructible_v<T>;
112 case constraint_level::nothrow:
113 return std::is_nothrow_copy_constructible_v<T>;
114 case constraint_level::trivial:
115 return std::is_trivially_copy_constructible_v<T> &&
116 std::is_trivially_destructible_v<T>;
117 default: return false;
118 }
119}
120template <class T>
121consteval bool has_relocatability(constraint_level level) {
122 switch (level) {
123 case constraint_level::none: return true;
124 case constraint_level::nontrivial:
125 return std::is_move_constructible_v<T> && std::is_destructible_v<T>;
126 case constraint_level::nothrow:
127 return std::is_nothrow_move_constructible_v<T> &&
128 std::is_nothrow_destructible_v<T>;
129 case constraint_level::trivial:
130 return std::is_trivially_move_constructible_v<T> &&
131 std::is_trivially_destructible_v<T>;
132 default: return false;
133 }
134}
135template <class T>
136consteval bool has_destructibility(constraint_level level) {
137 switch (level) {
138 case constraint_level::none: return true;
139 case constraint_level::nontrivial: return std::is_destructible_v<T>;
140 case constraint_level::nothrow: return std::is_nothrow_destructible_v<T>;
141 case constraint_level::trivial: return std::is_trivially_destructible_v<T>;
142 default: return false;

191template <class D, class P, qualifier_type Q, class R, class... Args>
192R indirect_conv_dispatcher(add_qualifier_t<std::byte, Q> self, Args... args)
193 noexcept(invocable_dispatch_ptr_indirect<D, P, Q, true, R, Args...>) {
194 return invoke_dispatch<D, R>(*std::forward<add_qualifier_t<P, Q>>(
195 *std::launder(reinterpret_cast<add_qualifier_ptr_t<P, Q>>(&self))),
196 std::forward<Args>(args)...);

198template <class D, class P, qualifier_type Q, class R, class... Args>
199R direct_conv_dispatcher(add_qualifier_t<std::byte, Q> self, Args... args)
200 noexcept(invocable_dispatch_ptr_direct<D, P, Q, true, R, Args...>) {
201 auto& qp = *std::launder(
202 reinterpret_cast<add_qualifier_ptr_t<P, Q>>(&self));
203 if constexpr (Q == qualifier_type::rv) {
204 destruction_guard guard{&qp};
205 return invoke_dispatch<D, R>(
206 std::forward<add_qualifier_t<P, Q>>(qp), std::forward<Args>(args)...);
207 } else {
208 return invoke_dispatch<D, R>(
209 std::forward<add_qualifier_t<P, Q>>(qp), std::forward<Args>(args)...);
210 }
211}

214 noexcept(invocable_dispatch<D, true, R, std::nullptr_t, Args...>)
215 { return invoke_dispatch<D, R>(nullptr, std::forward<Args>(args)...); }

218 noexcept(has_copyability<P>(constraint_level::nothrow)) {

223void copying_default_dispatcher(std::byte& self, const std::byte& rhs)
224 noexcept {

228template <class P>
229void relocation_dispatcher(std::byte& self, const std::byte& rhs)
230 noexcept(has_relocatability<P>(constraint_level::nothrow)) {
231 P* other = std::launder(reinterpret_cast<P*>(const_cast<std::byte*>(&rhs)));
232 destruction_guard guard{other};
233 std::construct_at(reinterpret_cast<P*>(&self), std::move(*other));
