743void Initializer::initBnds(Bnds* bnds, Equation* eqn, std::array<int, 3> iMax, Block* block)
744{
745 bnds->iMax = iMax;
746 bnds->dim = info->dim;
747
748 int nGhost = info->nGhostCell();
749 int nCons = info->nCons();
750 int nPrim = info->nPrim();
751
752 std::array<int, 3> nBnds { bnds->iMax[1] * bnds->iMax[2],
753 bnds->iMax[0] * bnds->iMax[2],
754 bnds->iMax[0] * bnds->iMax[1] };
755 int nBnd = 0;
756 for (int i = 0; i < bnds->dim; i++) {
757 nBnd += nBnds[i] * 2;
758 }
759
760 bnds->oneDBnds.resize(nBnd);
761 std::array<int, 2> offsets;
762 switch (info->eqType) {
763 case LINEARCONV1D:
764 case BURGERS1D: {
765 // 使用周期性边界条件
766 for (int i = 0; i < 2; i++) {
767 bnds->oneDBnds.at(i) = std::make_shared<OneDBnd>(nGhost, nPrim, PERIODIC1D);
768 }
769 offsets = calOffsetInverse(1, 0, 0, bnds->iMax);
770 bnds->oneDBnds.at(0)->setUpdate(eqn->prim, offsets[0], offsets[1]);
771
772 offsets = calOffset(1, 0, 0, bnds->iMax);
773 bnds->oneDBnds.at(1)->setUpdate(eqn->prim, offsets[0], offsets[1]);
774 } break;
775 case EULER:
776 if (eqn->dim == 1) {
777 // 一维欧拉方程边界条件
778 BndType Xtype = SUPERSONICOUTLET;
779 if (info->nCase == 4) {
780 Xtype = SYMMETRY1D;
781 }
782 bnds->oneDBnds.at(0) = std::make_shared<OneDBnd>(nGhost, nPrim, Xtype);
783 offsets = calOffset(1, 0, 0, bnds->iMax);
784 bnds->oneDBnds.at(0)->setUpdate(eqn->prim, offsets[0], offsets[1]);
785
786 bnds->oneDBnds.at(1) = std::make_shared<OneDBnd>(nGhost, nPrim, Xtype);
787 offsets = calOffsetInverse(1, 0, 0, bnds->iMax);
788 bnds->oneDBnds.at(1)->setUpdate(eqn->prim, offsets[0], offsets[1]);
789 } else if (eqn->dim == 2) {
790 // 二维欧拉方程边界条件
791 BndType Xtype = SUPERSONICOUTLET, Ytype = SUPERSONICOUTLET;
792 if (info->nCase == 2) {
793 Xtype = SYMMETRYX;
794 Ytype = SYMMETRYY;
795 }
796 if (info->nCase == 3) {
797 Xtype = SYMMETRYX;
798 Ytype = DIRICLET;
799 }
800 if (info->nCase == 4) {
801 // 双马赫反射问题需要特殊处理
802 initDoubleMachBnds(bnds, eqn, iMax, block);
803 break;
804 }
805
806 // 【王鸿飞】begin新算例KHI
807 if (info->nCase == 6) {
808 // 为KH不稳定性问题设置周期性边界条件
809 Xtype = PERIODIC1D;
810 Ytype = PERIODIC1D;
811 }
812 // 【王鸿飞】end
813
814 // X方向边界条件
815 for (int i = 0; i < iMax[1]; i++) {
816 bnds->oneDBnds.at(2 * i) = std::make_shared<OneDBnd>(nGhost, nPrim, Xtype);
817 offsets = calOffset(1, i, 0, bnds->iMax);
818 bnds->oneDBnds.at(2 * i)->setUpdate(eqn->prim, offsets[0], offsets[1]);
819
820 bnds->oneDBnds.at(2 * i + 1) = std::make_shared<OneDBnd>(nGhost, nPrim, Xtype);
821 offsets = calOffsetInverse(1, i, 0, bnds->iMax);
822 bnds->oneDBnds.at(2 * i + 1)->setUpdate(eqn->prim, offsets[0], offsets[1]);
823 }
824
825 // Y方向边界条件
826 for (int i = 0; i < iMax[0]; i++) {
827 bnds->oneDBnds.at(2 * i + iMax[1] * 2) = std::make_shared<OneDBnd>(nGhost, nPrim, Ytype);
828 offsets = calOffset(2, i, 0, bnds->iMax);
829 bnds->oneDBnds.at(2 * i + iMax[1] * 2)->setUpdate(eqn->prim, offsets[0], offsets[1]);
830
831 if (info->nCase == 3) // For RTI
832 {
833 // 为Rayleigh-Taylor不稳定性问题设置Dirichlet边界条件值
834 std::array<real, 4> dirVar = { 2.0, 0, 0, 1.0 };
835 std::vector<real> dirVars(nGhost * nPrim);
836 for (int j = 0; j < nGhost; j++) {
837 dirVars.at(j * nPrim + 0) = dirVar[0];
838 dirVars.at(j * nPrim + 1) = dirVar[1];
839 dirVars.at(j * nPrim + 2) = dirVar[2];
840 dirVars.at(j * nPrim + 3) = dirVar[3];
841 }
842 bnds->oneDBnds.at(2 * i + iMax[1] * 2)->setValue(dirVars);
843 }
844
845 bnds->oneDBnds.at(2 * i + 1 + iMax[1] * 2) = std::make_shared<OneDBnd>(nGhost, nPrim, Ytype);
846 offsets = calOffsetInverse(2, i, 0, bnds->iMax);
847 bnds->oneDBnds.at(2 * i + 1 + iMax[1] * 2)->setUpdate(eqn->prim, offsets[0], offsets[1]);
848
849 if (info->nCase == 3) // For RTI
850 {
851 // 为Rayleigh-Taylor不稳定性问题设置Dirichlet边界条件值
852 std::array<real, 4> dirVar = { 1.0, 0, 0, 2.5 };
853 std::vector<real> dirVars(nGhost * nPrim);
854 for (int j = 0; j < nGhost; j++) {
855 dirVars.at(j * nPrim + 0) = dirVar[0];
856 dirVars.at(j * nPrim + 1) = dirVar[1];
857 dirVars.at(j * nPrim + 2) = dirVar[2];
858 dirVars.at(j * nPrim + 3) = dirVar[3];
859 }
860 bnds->oneDBnds.at(2 * i + iMax[1] * 2 + 1)->setValue(dirVars);
861 }
862 }
863 }
864
865 break;
866
867 default:
868 break;
869 }
