229std::string Info::filename()
230{
231 // 字符串拼接，包含空间离散方法、插值方法、算例信息和当前时间
232 std::string caseName = "unknown";
233
234 if (dim == 1) {
235 auto it = exampleStr1D.find(nCase);
236 if (it != exampleStr1D.end()) {
237 caseName = it->second;
238 }
239 } else if (dim == 2) {
240 auto it = exampleStr2D.find(nCase);
241 if (it != exampleStr2D.end()) {
242 caseName = it->second;
243 }
244 }
245
246 // 添加网格信息到文件名
247 std::string gridInfo = "";
248 if (dim == 1) {
249 gridInfo = std::to_string(iMax[0]);
250 } else if (dim == 2) {
251 gridInfo = std::to_string(iMax[0]) + "x" + std::to_string(iMax[1]);
252 } else if (dim == 3) {
253 gridInfo = std::to_string(iMax[0]) + "x" + std::to_string(iMax[1]) + "x" + std::to_string(iMax[2]);
254 }
255
256 return caseName + " - " + disStr[interMethod] + " - " + gridInfo + " - " + std::format("t={:.4f}.cgns",t);
