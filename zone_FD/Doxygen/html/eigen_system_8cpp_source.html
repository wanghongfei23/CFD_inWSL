222 real rr = primr[0], ur = primr[1], pr = primr[2];
223
224 std::array<real, 2> H;
225 H[LL] = (ul * ul) / 2 + pl / rl * gamma / (gamma - 1);
226 H[RR] = (ur * ur) / 2 + pr / rr * gamma / (gamma - 1);

232 u = (coef1 * ul + coef2 * ur) * divisor;
233 real ht = (coef1 * H[LL] + coef2 * H[RR]) * divisor;
234 ek = (u * u) / 2;
235 h = ht - ek;
236 c = std::sqrt((gamma - 1) * h);
237 p = r * h * ((gamma - 1) / gamma);
238
239 real gamma_1 = gamma - 1;
240 leftEig = {ht + c * (u - c) / gamma_1, -u - c / gamma_1, 1.0,
241 -2.0 * ht + 4.0 * c * c / gamma_1, 2.0 * u, -2.0,
242 ht - c * (u + c) / gamma_1, -u + c / gamma_1, 1.0};
243 real factorEig = 0.5 * gamma_1 / (c * c);
244
245 for (int ii = 0; ii < 9; ii++)
246 leftEig[ii] *= factorEig;
247
248 rightEig = {1.0, 1.0, 1.0, u - c, u,
249 u + c, ht - u * c, 0.5 * u * u, ht + u * c};
