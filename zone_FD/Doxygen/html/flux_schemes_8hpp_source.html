13inline void roeFlux1D2(const std::span<real, 6>& iter,
14 const std::span<real, 3>& res,
15 const std::array<real, 3>& norm)
16{
17 // reference: https://blog.csdn.net/Tankrun1997/article/details/132743487
18 real rl = iter[0];
19 real ul = iter[1];
20 real pl = iter[2];
21 real rr = iter[3];
22 real ur = iter[4];
23 real pr = iter[5];
24 enum { L,
25 R };
26
27 real gamma = GAMMA;
28
29 arr2 H = { pl / rl * GAMMA / (GAMMA - 1) + (ul * ul) / 2,
30 pr / rr * GAMMA / (GAMMA - 1) + (ur * ur) / 2 };
31
32 real rBar = std::sqrt(rl * rr);
33 real uBar = (ul * std::sqrt(rl) + ur * std::sqrt(rr)) / (std::sqrt(rl) + std::sqrt(rr));
34 real HBar = (H[L] * std::sqrt(rl) + H[R] * std::sqrt(rr)) / (std::sqrt(rl) + std::sqrt(rr));
35 real cBar = std::sqrt((gamma - 1) * (HBar - uBar * uBar / 2));
36 real cBar2 = (gamma - 1) * (HBar - uBar * uBar / 2);
37
38 real dr = rr - rl;
39 real du = ur - ul;
40 real dp = pr - pl;
41
42 real K1[3] = { 1, uBar, uBar * uBar / 2 };
43 real K2[3] = { 1, uBar - cBar, HBar - uBar * cBar };
44 real K3[3] = { 1, uBar + cBar, HBar + uBar * cBar };
45
46 real lambda[3] = { std::abs(uBar), std::abs(uBar + cBar),
47 std::abs(uBar - cBar) };
48
49 real eps = 0.1 * (std::abs(uBar) + cBar);
50 for (int i = 0; i < 3; i++) {
51 if (lambda[i] < eps) {
52 lambda[i] = (lambda[i] * lambda[i] + eps * eps) / (2.0 * eps);
53 }
54 }
55
56 real alpha[5] = { lambda[0] * (dr - dp / cBar2),
57 lambda[1] / (2.0 * cBar2) * (dp + rBar * cBar * du),
58 lambda[2] / (2.0 * cBar2) * (dp - rBar * cBar * du) };
59 alpha[3] = alpha[0] + alpha[1] + alpha[2];
60 alpha[4] = cBar * (alpha[1] - alpha[2]);
61
62 real FL[3] = { rl * ul, rl * ul * ul + pl, rl * H[L] * ul };
63 real FR[3] = { rr * ur, rr * ur * ur + pr, rr * H[R] * ur };
64
65 for (int i = 0; i < 3; i++) {
66 res[i] = (FL[i] + FR[i]) / 2.0;
67 }
68 res[0] -= 0.5 * alpha[3];
69 res[1] -= 0.5 * (alpha[3] * uBar + alpha[4]);
70 res[2] -= 0.5 * (HBar * alpha[3] + uBar * alpha[4] - cBar2 * alpha[0] / (gamma - 1));
71}

82 real rr = iter[3];
83 real ur = iter[4];
84 real pr = iter[5];

86 R };
87 arr2 H = { pl / rl * GAMMA / (GAMMA - 1) + (ul * ul) / 2,
88 pr / rr * GAMMA / (GAMMA - 1) + (ur * ur) / 2 };
89 arr2 RT = { pl / rl, pr / rr };
90 real gamma = GAMMA;
91 real cl = std::sqrt(gamma * RT[L]);
92 real cr = std::sqrt(gamma * RT[R]);
93
94 real uBar = (ul + ur) / 2, cBar = (cl + cr) / 2;
95 real SL = std::min(ur - cr, ul - cl);
96 real SR = std::max(ul + cl, ur + cr);
97
98 real Sstar = ((pr - pl) + (rl * ul * (SL - ul) - rr * ur * (SR - ur))) / (rl * (SL - ul) - rr * (SR - ur));
99 if (SL >= 0) {
100 res[0] = rl * ul;
101 res[1] = rl * ul * ul + pl;
102 res[2] = rl * H[L] * ul;
103 } else if (Sstar >= 0) {
104 real pStar = pl + rl * (SL - ul) * (Sstar - ul);
105 real U[3] = { rl, rl * ul, pl / (gamma - 1) + rl * ul * ul / 2 };
106 real F[3] = { rl * ul, rl * ul * ul + pl, rl * H[L] * ul };
107 real D[3] = { 0, 1, Sstar };
108 for (int i = 0; i < 3; i++) {
109 res[i] = (Sstar * (SL * U[i] - F[i]) + SL * pStar * D[i]) / (SL - Sstar);
110 }
111
112 } else if (SR >= 0) {
113 real pStar = pr + rr * (SR - ur) * (Sstar - ur);
114 real U[3] = { rr, rr * ur, pr / (gamma - 1) + rr * ur * ur / 2 };
115 real F[3] = { rr * ur, rr * ur * ur + pr, rr * H[R] * ur };
116 real D[3] = { 0, 1, Sstar };
117 for (int i = 0; i < 3; i++) {
118 res[i] = (Sstar * (SR * U[i] - F[i]) + SR * pStar * D[i]) / (SR - Sstar);
119 }
120 } else {
121 res[0] = rr * ur;
122 res[1] = rr * ur * ur + pr;
123 res[2] = rr * H[R] * ur;
124 }
125}

329// 二维Roe通量计算函数
330inline void roeFlux2D(const std::span<real, 8>& iter,
331 const std::span<real, 4>& res,
332 const std::array<real, 3>& norm)
333{
334
335 real rl = iter[0];
336 real ul = iter[1];
337 real vl = iter[2];
338 real pl = iter[3];
339 real rr = iter[4];
340 real ur = iter[5];
341 real vr = iter[6];
342 real pr = iter[7];
343 enum { L,
344 R };
345 std::array<real, 4> FcL, FcR;
346 if (pl < 0 || pr < 0) {
347 std::cout << "fluxScheme error: Pressure positivity break\n";
348 }
349 if (rr < 0 || rl < 0) {
350 std::cout << "fluxScheme error: Density positivity break\n";
351 }
352
353 arr2 H;
354 H[L] = (ul * ul + vl * vl) / 2 + pl / rl * GAMMA / (GAMMA - 1);
355 H[R] = (ur * ur + vr * vr) / 2 + pr / rr * GAMMA / (GAMMA - 1);
356
357 arr2 Vn = { ul * norm[0] + vl * norm[1], ur * norm[0] + vr * norm[1] };
358 FcL[0] = rl * Vn[L];
359 FcL[1] = ul * Vn[L] * rl + pl * norm[0];
360 FcL[2] = vl * Vn[L] * rl + pl * norm[1];
361 FcL[3] = rl * H[L] * Vn[L];
362
363 FcR[0] = rr * Vn[R];
364 FcR[1] = ur * Vn[R] * rr + pr * norm[0];
365 FcR[2] = vr * Vn[R] * rr + pr * norm[1];
366 FcR[3] = rr * H[R] * Vn[R];
367
368 double rhoAvg, uAvg, vAvg, HAvg, cAvg, VnAvg, q_2Avg, coef1, coef2;
369 coef1 = std::sqrt(rl);
370 coef2 = std::sqrt(rr);
371 real divisor = 1.0 / (std::sqrt(rl) + std::sqrt(rr));
372 rhoAvg = std::sqrt(rl * rr);
373 uAvg = (coef1 * ul + coef2 * ur) * divisor;
374 vAvg = (coef1 * vl + coef2 * vr) * divisor;
375 HAvg = (coef1 * H[L] + coef2 * H[R]) * divisor;
376 q_2Avg = (uAvg * uAvg + vAvg * vAvg) / 2;
377
378 cAvg = std::sqrt((GAMMA - 1) * (HAvg - q_2Avg));
379 if ((HAvg - q_2Avg) < 0) {
380 cAvg = std::sqrt(0.1);
381 }
382 VnAvg = uAvg * norm[0] + vAvg * norm[1];
383
384 double lambda[3] = { std::abs(VnAvg - cAvg), std::abs(VnAvg),
385 std::abs(VnAvg + cAvg) };
386 real eps = 0.05 * (std::abs(VnAvg) + cAvg);
387 for (int i = 0; i < 3; i++) {
388 if (lambda[i] < eps) {
389 lambda[i] = (lambda[i] * lambda[i] + eps * eps) / (2.0 * eps);
390 }
391 }
392
393 double deltaP = pr - pl, deltaVn = Vn[R] - Vn[L], deltaU = ur - ul,
394 deltaV = vr - vl, deltaRho = rr - rl, coef;
395 double FDispassion[4];
396 coef1 = (deltaP - rhoAvg * cAvg * deltaVn) / (2 * cAvg * cAvg);
397 FDispassion[0] = lambda[0] * coef1 * 1;
398 FDispassion[1] = lambda[0] * coef1 * (uAvg - cAvg * norm[0]);
399 FDispassion[2] = lambda[0] * coef1 * (vAvg - cAvg * norm[1]);
400 FDispassion[3] = lambda[0] * coef1 * (HAvg - cAvg * VnAvg);
401
402 coef2 = deltaRho - deltaP / (cAvg * cAvg);
403 FDispassion[0] += lambda[1] * (coef2 * 1.0 + rhoAvg * 0.0);
404 FDispassion[1] += lambda[1] * (coef2 * uAvg + rhoAvg * (deltaU - deltaVn * norm[0]));
405 FDispassion[2] += lambda[1] * (coef2 * vAvg + rhoAvg * (deltaV - deltaVn * norm[1]));
406 FDispassion[3] += lambda[1] * (coef2 * q_2Avg + rhoAvg * (uAvg * deltaU + vAvg * deltaV - VnAvg * deltaVn));
407
408 coef = (deltaP + rhoAvg * cAvg * deltaVn) / (2.0 * cAvg * cAvg);
409 FDispassion[0] += lambda[2] * (coef * 1);
410 FDispassion[1] += lambda[2] * (coef * (uAvg + cAvg * norm[0]));
411 FDispassion[2] += lambda[2] * (coef * (vAvg + cAvg * norm[1]));
412 FDispassion[3] += lambda[2] * (coef * (HAvg + cAvg * VnAvg));
413
414 for (int i = 0; i < 4; i++) {
415 res[i] = 0.5 * (FcL[i] + FcR[i] - FDispassion[i]);
416 }
417}
