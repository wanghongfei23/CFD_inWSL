43void BlockSolver::RK3_SSP(real dt)
44{
45
46 Data tempdata(*cons);
47 int n = cons->size();
48
49 // third order RK
50 // stage 1
51 rhs->setZeros();
52 eqn->consToPrim();
53 bnds->update();
54 spDis->rhsSolve();
55 sourceTerm->calSource();
56// cgnsIO.BlockCgnsOutput(block,info);
57// cgnsIO.solCgnsOutput(rhs,info);
58#pragma omp parallel for
59 for (int i = 0; i < n; i++) {
60 (*cons)[i] = tempdata[i] - dt * (*rhs)[i];
61 }
62 info->t += dt;
63
64 // stage 2
65 rhs->setZeros();
66 eqn->consToPrim();
67 bnds->update();
68 spDis->rhsSolve();
69 sourceTerm->calSource();
70#pragma omp parallel for
71 for (int i = 0; i < n; i++) {
72 (*cons)[i] = 0.75 * tempdata[i] - 0.25 * dt * (*rhs)[i] + 0.25 * (*cons)[i];
73 }
74 info->t -= dt / 2;
75
76 // stage 3
77 rhs->setZeros();
78 eqn->consToPrim();
79 bnds->update();
80 spDis->rhsSolve();
81 sourceTerm->calSource();
82#pragma omp parallel for
83 for (int i = 0; i < n; i++) {
84 (*cons)[i] = 1.0 / 3.0 * tempdata[i] - 2.0 / 3.0 * dt * (*rhs)[i] + 2.0 / 3.0 * (*cons)[i];
85 }
86 info->t -= dt / 2;
87}

89void BlockSolver::DTS_Euler(real dt)
90{
91
92 int n = cons->size();
93 Data tempdata(*cons);
94 Data tempRhs(*cons);
95 real maxres = 0;
96
97 int imStep = 0;
98 do {
99
100 auto dtau = calLocalCFL();
101 rhs->setZeros();
102 eqn->consToPrim();
103 bnds->update();
104 spDis->rhsSolve();
105 sourceTerm->calSource();
106 // cgnsIO.BlockCgnsOutput(block,info);
107 // cgnsIO.solCgnsOutput(rhs,info);
108 tempRhs.setZeros();
109#pragma omp parallel for
110 for (int i = 0; i < n; i++) {
111 real coef = dtau[i] / (dt + dtau[i]);
112 tempRhs[i] = (tempdata[i] - (*cons)[i] - dt * (*rhs)[i]) * coef;
113 (*cons)[i] = (*cons)[i] + tempRhs[i];
114 }
115 maxres = tempRhs.getLinf(0);
116 std::cout << std::format("time = {:.4f} dt={:.10f} imStep={} rhoL2 = {} \n", info->t, dt, imStep, maxres);
117 } while (imStep++ < info->maxImplicitStep && maxres > 1e-7);
118}

170void BlockSolver::stepsLoopCFL()
171{
172 // 定义一个极小值，用于浮点数比较
173 const real eps = 1e-15;
174
175 // 时间步进循环，直到达到结束步数
176 while (info->step < info->endStep) {
177 // 当达到输出时间时，输出网格和物理量信息
178 if (info->outputT < eps) {
179 outputGrid();
180 outputPrim();
181 }
182
183 // 根据CFL条件计算时间步长
184 auto dt = getTimeIntervalExplicit();
185 // dt=info->dt;
186
187 // 检查是否需要调整时间步长以匹配输出间隔
188 if (info->outputT + dt > info->outputDt) {
189 dt = info->outputDt - info->outputT;
190 info->outputT = 0.0;
191 info->step++;
192 } else
193 info->outputT += dt;
194
195 // 执行一次时间步进求解并计时
196 auto start = std::chrono::high_resolution_clock::now();
197 solve(dt);
198 auto stop = std::chrono::high_resolution_clock::now();
199 auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(stop - start).count();
200 timesss += duration;
201 timesteps++;
202
203 // 输出当前时间、时间步长和密度残差的L∞范数
204 std::string output_str = std::format("time = {:.4f} dt={:.10f} rhoLinf = {:.4f} \n", info->t, dt, rhs->getLinf(0));
205
206 std::cout << output_str;
207
208 // 【王鸿飞】begin辅助工具
209
210 // 将相同信息写入txt文件，使用与CGNS文件相同的命名方式，但不包含时间戳
211 std::string base_filename = info->filename();
212 // 移除时间戳部分（从最后一个" - t="开始的部分）
213 size_t pos = base_filename.rfind(" - t=");
214 if (pos != std::string::npos) {
215 base_filename = base_filename.substr(0, pos) + " - rholinf.txt";
216 } else {
217 // 如果没有找到时间戳模式，则直接替换扩展名
218 pos = base_filename.find(".cgns");
219 if (pos != std::string::npos) {
220 base_filename.replace(pos, 5, ".txt");
221 }
222 }
223
224 std::ofstream log_file(base_filename, std::ios::app);
225 if (log_file.is_open()) {
226 log_file << output_str;
227 log_file.close();
228 }
229 // 【王鸿飞】end辅助工具
230 }
231
232 // 循环结束后最后输出一次结果
233 outputGrid();
234 outputPrim();
235}

247std::vector<real> BlockSolver::calLocalCFL()
248{
249
250 real dt, lambda = 0;
251
252 Data* prim = eqn->getPrim();
253 int n = prim->getN();
254 std::vector<real> res(n);
255 eqn->consToPrim();
256#pragma omp parallel for reduction(max : lambda)
257 for (int i = 0; i < n; i++) {
258 real iLambda;
259 real dh = block->getMinDh(i);
260 real RT = std::sqrt(GAMMA * (*prim)(i, 2) / (*prim)(i, 0)) + std::abs((*prim)(i, 1));
261 if (info->dim == 2)
262 iLambda = (std::sqrt(GAMMA * (*prim)(i, 3) / (*prim)(i, 0)) + std::abs((*prim)(i, 1)) + std::abs((*prim)(i, 2))) / dh;
263 else if (info->dim == 1)
264 iLambda = (std::sqrt(GAMMA * (*prim)(i, 2) / (*prim)(i, 0)) + std::abs((*prim)(i, 1))) / dh;
265 res[i] = info->implicitCFL / iLambda;
266 }
267 if (info->eqType != EULER) {

318real BlockSolver::getTimeIntervalExplicit()
319{
320 real dt, lambda = 0;
321 int dim = info->dim;
322 if (info->eqType == EULER) {
323 Data* prim = eqn->getPrim();
324 int n = prim->getN();
325 eqn->consToPrim();
326 std::vector<real> lambdas(dim);
327 for (int idim = 0; idim < dim; idim++) {
328 real maxLambda = 0;
329#pragma omp parallel for reduction(max : maxLambda)
330 for (int i = 0; i < n; i++) {
331 real iLambda;
332 auto dhs = block->getCellInterval(i);
333 real dh = info->constH ? info->geth(idim) : dhs.at(idim);
334 iLambda = (std::sqrt(GAMMA * (*prim)(i, dim + 1) / (*prim)(i, 0)) + std::abs((*prim)(i, idim + 1))) / dh;
335 maxLambda = std::max(maxLambda, iLambda);
336 }
337 lambdas[idim] = maxLambda;
338 }
339 lambda = 0;
340 for (auto iLambda : lambdas)
341 lambda += iLambda;
342
343 } else {
344 std::cout << "Block Solver error: CFL Loop with not EULER solver\n";
345 lambda = 1;
346 }
347 dt = info->CFL / lambda;
348 return dt;
