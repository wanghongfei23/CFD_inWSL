99用于边界数据访问，可以实现不同的边界格式，比如周期性或者对称
100
101计算三维数据在指定方向上的索引偏移量
102输入：方向idim (1-y,2-z,3-x); 坐标i,j; 各维数组大小iMax
103输出：返回一个包含起始索引和偏移量的数组
104*/
105inline std::array<int, 2> calOffset(int idim, int i, int j, std::array<int, 3> iMax)
106{
107 std::array<int, 3> offsets { 1, iMax[0], iMax[0] * iMax[1] };
108 std::array<int, 2> res;
109 if (idim == 1) {
110 res[0] = i * offsets[1] + j * offsets[2]; // i0
111 res[1] = offsets[0]; // offset
112 } else if (idim == 2) {
113 res[0] = i * offsets[0] + j * offsets[2]; // i0
114 res[1] = offsets[1]; // offset
115 } else if (idim == 3) {
116 res[0] = i * offsets[0] + j * offsets[1]; // i0
117 res[1] = offsets[2]; // offset
118 }
119 return res;
120}
121// 计算三维数据在指定方向上的索引偏移量（反向）
122inline std::array<int, 2> calOffsetInverse(int idim, int i, int j,std::array<int, 3> iMax)
123{
124 std::array<int, 3> offsets { 1, iMax[0], iMax[0] * iMax[1] };
125 std::array<int, 2> res;
126 if (idim == 1) {
127 res[0] = i * offsets[1] + j * offsets[2] + (iMax[0] - 1) * offsets[0]; // i0
128 res[1] = -offsets[0]; // offset
129 } else if (idim == 2) {
130 res[0] = i * offsets[0] + j * offsets[2] + (iMax[1] - 1) * offsets[1]; // i0
131 res[1] = -offsets[1]; // offset
132 } else if (idim == 3) {
133 res[0] = i * offsets[0] + j * offsets[1] + (iMax[2] - 1) * offsets[2]; // i0
134 res[1] = -offsets[2]; // offset
135 }
136 return res;
